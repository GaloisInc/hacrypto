3 /*
4  * Description:
5  * This file implements the Secure Hash Signature Standard
6  * algorithms as defined in the National Institute of Standards
7  * and Technology Federal Information Processing Standards
8  * Publication (FIPS PUB) 180-1 published on April 17, 1995, 180-2
9  * published on August 1, 2002, and the FIPS PUB 180-2 Change
10  * Notice published on February 28, 2004.
11  *
12  * A combined document showing all algorithms is available at
13  * http://csrc.nist.gov/publications/fips/
14  * fips180-2/fips180-2withchangenotice.pdf
15  *
16  * The SHA-224 and SHA-256 algorithms produce 224-bit and 256-bit
17  * message digests for a given data stream. It should take about
18  * 2**n steps to find a message with the same digest as a given
19  * message and 2**(n/2) to find any two messages with the same
20  * digest, when n is the digest size in bits. Therefore, this
21  * algorithm can serve as a means of providing a
22  * "fingerprint" for a message.
23  *
24  * Portability Issues:
25  * SHA-224 and SHA-256 are defined in terms of 32-bit "words".
26  * This code uses <stdint.h> (included via "sha.h") to define 32
27  * and 8 bit unsigned integer types. If your C compiler does not
28  * support 32 bit unsigned integers, this code is not
29  * appropriate.
30  *
31  * Caveats:
32  * SHA-224 and SHA-256 are designed to work with messages less
33  * than 2^64 bits long. This implementation uses SHA224/256Input()
34  * to hash the bits that are a multiple of the size of an 8-bit
35  * character, and then uses SHA224/256FinalBits() to hash the
36  * final few bits of the input.
37  */

276  /* 0 0b00000000 */ 0x00, /* 1 0b10000000 */ 0x80,
277  /* 2 0b11000000 */ 0xC0, /* 3 0b11100000 */ 0xE0,
278  /* 4 0b11110000 */ 0xF0, /* 5 0b11111000 */ 0xF8,
279  /* 6 0b11111100 */ 0xFC, /* 7 0b11111110 */ 0xFE
280  };

283  /* 0 0b10000000 */ 0x80, /* 1 0b01000000 */ 0x40,
284  /* 2 0b00100000 */ 0x20, /* 3 0b00010000 */ 0x10,
285  /* 4 0b00001000 */ 0x08, /* 5 0b00000100 */ 0x04,
286  /* 6 0b00000010 */ 0x02, /* 7 0b00000001 */ 0x01
287  };

290  return shaSuccess;
291 
292  if (!context)
293  return shaNull;
294 
295  if ((context->Computed) || (length >= 8) || (length == 0)) {
296  context->Corrupted = shaStateError;
297  return shaStateError;
298  }

301  return context->Corrupted;

365 /*
366  * SHA224_256PadMessage
367  *
368  * Description:
369  * According to the standard, the message must be padded to an
370  * even 512 bits. The first padding bit must be a '1'. The
371  * last 64 bits represent the length of the original message.
372  * All bits in between should be 0. This helper function will pad
373  * the message according to those rules by filling the
374  * Message_Block array accordingly. When it returns, it can be
375  * assumed that the message digest has been computed.
376  *
377  * Parameters:
378  * context: [in/out]
379  * The context to pad
380  * Pad_Byte: [in]
381  * The last byte to add to the digest before the 0-padding
382  * and length. This will contain the last bits of the message
383  * followed by another single bit. If the message was an
384  * exact multiple of 8-bits long, Pad_Byte will be 0x80.
385  *
386  * Returns:
387  * Nothing.
388  */
389 static void SHA224_256PadMessage(SHA256Context *context,
390  uint8_t Pad_Byte)
391 {
392  /*
393  * Check to see if the current message block is too small to hold
394  * the initial padding bits and length. If so, we will pad the
395  * block, process it, and then continue padding into a second
396  * block.
397  */
398  if (context->Message_Block_Index >= (SHA256_Message_Block_Size-8)) {
399  context->Message_Block[context->Message_Block_Index++] = Pad_Byte;
400  while (context->Message_Block_Index < SHA256_Message_Block_Size)
401  context->Message_Block[context->Message_Block_Index++] = 0;
402  SHA224_256ProcessMessageBlock(context);
403  } else
404  context->Message_Block[context->Message_Block_Index++] = Pad_Byte;
405 
406  while (context->Message_Block_Index < (SHA256_Message_Block_Size-8))
407  context->Message_Block[context->Message_Block_Index++] = 0;
408 
409  /*
410  * Store the message length as the last 8 octets
411  */
412  context->Message_Block[56] = (uint8_t)(context->Length_High >> 24);
413  context->Message_Block[57] = (uint8_t)(context->Length_High >> 16);
414  context->Message_Block[58] = (uint8_t)(context->Length_High >> 8);
415  context->Message_Block[59] = (uint8_t)(context->Length_High);
416  context->Message_Block[60] = (uint8_t)(context->Length_Low >> 24);
417  context->Message_Block[61] = (uint8_t)(context->Length_Low >> 16);
418  context->Message_Block[62] = (uint8_t)(context->Length_Low >> 8);
419  context->Message_Block[63] = (uint8_t)(context->Length_Low);
420 
421  SHA224_256ProcessMessageBlock(context);
422 }
