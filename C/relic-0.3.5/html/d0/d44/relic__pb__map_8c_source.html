43 void pb_map_init() {
44  ctx_t *ctx = core_get();
45 
46  for (int i = 0; i < FB_TABLE; i++) {
47  ctx->pb_ptr[i] = &(ctx->pb_exp[i]);
48  }
49 
50 #if PB_MAP == ETATS || PB_MAP == ETATN
51  fb_itr_pre((fb_t *)pb_map_get_tab(), 4 * (((FB_BITS + 1) / 2) / 4));
52 #endif
53 
54 #if defined(PB_PARAL) && (PB_MAP == ETATS || PB_MAP == ETATN)
55  int chunk = (int)ceilf((FB_BITS - 1) / (2.0 * CORES));
56 
57  for (int i = 0; i < CORES; i++) {
58  for (int j = 0; j < FB_TABLE; j++) {
59  ctx->pb_pow[0][i][j] = &(ctx->pb_sqr[i][j]);
60  ctx->pb_pow[1][i][j] = &(ctx->pb_srt[i][j]);
61  }
62  }
63 
64  for (int i = 0; i < CORES; i++) {
65  fb_itr_pre(pb_map_get_sqr(i), i * chunk);
66  fb_itr_pre(pb_map_get_srt(i), -i * chunk);
67  }
68 
69 #endif
70 }
