39 int fp2_srt(fp2_t c, fp2_t a) {
40  int r = 0;
41  fp_t t1;
42  fp_t t2;
43  fp_t t3;
44 
45  fp_null(t1);
46  fp_null(t2);
47  fp_null(t3);
48 
49  TRY {
50  fp_new(t1);
51  fp_new(t2);
52  fp_new(t3);
53 
54  /* t1 = a[0]^2 - u^2 * a[1]^2 */
55  fp_sqr(t1, a[0]);
56  fp_sqr(t2, a[1]);
57  for (int i = -1; i > fp_prime_get_qnr(); i--) {
58  fp_add(t1, t1, t2);
59  }
60  fp_add(t1, t1, t2);
61 
62  if (fp_srt(t2, t1)) {
63  /* t1 = (a_0 + sqrt(t1)) / 2 */
64  fp_add(t1, a[0], t2);
65  fp_set_dig(t3, 2);
66  fp_inv(t3, t3);
67  fp_mul(t1, t1, t3);
68 
69  if (!fp_srt(t3, t1)) {
70  /* t1 = (a_0 - sqrt(t1)) / 2 */
71  fp_sub(t1, a[0], t2);
72  fp_set_dig(t3, 2);
73  fp_inv(t3, t3);
74  fp_mul(t1, t1, t3);
75  fp_srt(t3, t1);
76  }
77  /* c_0 = sqrt(t1) */
78  fp_copy(c[0], t3);
79  /* c_1 = a_1 / (2 * sqrt(t1)) */
80  fp_dbl(t3, t3);
81  fp_inv(t3, t3);
82  fp_mul(c[1], a[1], t3);
83  r = 1;
84  }
85  }
86  CATCH_ANY {
87  THROW(ERR_CAUGHT);
88  }
89  FINALLY {
90  fp_free(t1);
91  fp_free(t2);
92  fp_free(t3);
93  }
94  return r;
95 }
