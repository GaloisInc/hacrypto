135 static void bn_exp(bn_t c, const bn_t a, const bn_t b, const bn_t m) {
136  int i, l;
137  bn_t t;
138 
139  bn_null(t);
140 
141  TRY {
142  bn_new(t);
143 
144  l = bn_bits(b);
145 
146  bn_copy(t, a);
147 
148  for (i = l - 2; i >= 0; i--) {
149  bn_sqr(t, t);
150  bn_mod(t, t, m);
151  if (bn_test_bit(b, i)) {
152  bn_mul(t, t, a);
153  bn_mod(t, t, m);
154  }
155  }
156 
157  bn_copy(c, t);
158  }
159  CATCH_ANY {
160  THROW(ERR_CAUGHT);
161  }
162  FINALLY {
163  bn_free(t);
164  }
165 }

197 int bn_is_prime_basic(const bn_t a) {
198  dig_t t;
199  int i, result;
200 
201  result = 1;
202 
203  if (bn_cmp_dig(a, 1) == CMP_EQ) {
204  return 0;
205  }
206 
207  /* Trial division. */
208  for (i = 0; i < BASIC_TESTS; i++) {
209  bn_mod_dig(&t, a, primes[i]);
210  if (t == 0 && bn_cmp_dig(a, primes[i]) != CMP_EQ) {
211  result = 0;
212  break;
213  }
214  }
215  return result;
216 }

218 int bn_is_prime_rabin(const bn_t a) {
219  bn_t t, n1, y, r;
220  int i, s, j, result, b, tests = 0;
221 
222  tests = 0;
223  result = 1;
224 
225  bn_null(t);
226  bn_null(n1);
227  bn_null(y);
228  bn_null(r);
229 
230  if (bn_cmp_dig(a, 1) == CMP_EQ) {
231  return 0;
232  }
233 
234  TRY {
235  /*
236  * These values are taken from Table 4.4 inside Handbook of Applied
237  * Cryptography.
238  */
239  b = bn_bits(a);
240  if (b >= 1300) {
241  tests = 2;
242  } else if (b >= 850) {
243  tests = 3;
244  } else if (b >= 650) {
245  tests = 4;
246  } else if (b >= 550) {
247  tests = 5;
248  } else if (b >= 450) {
249  tests = 6;
250  } else if (b >= 400) {
251  tests = 7;
252  } else if (b >= 350) {
253  tests = 8;
254  } else if (b >= 300) {
255  tests = 9;
256  } else if (b >= 250) {
257  tests = 12;
258  } else if (b >= 200) {
259  tests = 15;
260  } else if (b >= 150) {
261  tests = 18;
262  } else {
263  tests = 27;
264  }
265 
266  bn_new(t);
267  bn_new(n1);
268  bn_new(y);
269  bn_new(r);
270 
271  /* r = (n - 1)/2^s. */
272  bn_sub_dig(n1, a, 1);
273  s = 0;
274  while (bn_is_even(n1)) {
275  s++;
276  bn_rsh(n1, n1, 1);
277  }
278  bn_lsh(r, n1, s);
279 
280  for (i = 0; i < tests; i++) {
281  /* Fix the basis as the first few primes. */
282  bn_set_dig(t, primes[i]);
283 
284  /* y = b^r mod a. */
285 #if BN_MOD != PMERS
286  bn_mxp(y, t, r, a);
287 #else
288  bn_exp(y, t, r, a);
289 #endif
290 
291  if (bn_cmp_dig(y, 1) != CMP_EQ && bn_cmp(y, n1) != CMP_EQ) {
292  j = 1;
293  while ((j <= (s - 1)) && bn_cmp(y, n1) != CMP_EQ) {
294  bn_sqr(y, y);
295  bn_mod(y, y, a);
296 
297  /* If y == 1 then composite. */
298  if (bn_cmp_dig(y, 1) == CMP_EQ) {
299  result = 0;
300  break;
301  }
302  ++j;
303  }
304 
305  /* If y != n1 then composite. */
306  if (bn_cmp(y, n1) != CMP_EQ) {
307  result = 0;
308  break;
309  }
310  }
311  }
312  }
313  CATCH_ANY {
314  result = 0;
315  THROW(ERR_CAUGHT);
316  }
317  FINALLY {
318  bn_free(r);
319  bn_free(y);
320  bn_free(n1);
321  bn_free(t);
322  }
323  return result;
324 }

326 int bn_is_prime_solov(const bn_t a) {
327  bn_t t0, t1, t2;
328  int i, result;
329 
330  bn_null(t0);
331  bn_null(t1);
332  bn_null(t2);
333 
334  result = 1;
335 
336  TRY {
337  bn_new(t0);
338  bn_new(t1);
339  bn_new(t2);
340 
341  for (i = 0; i < 100; i++) {
342  /* Generate t0, 2 <= t0, <= a - 2. */
343  do {
344  bn_rand(t0, BN_POS, bn_bits(a));
345  bn_mod(t0, t0, a);
346  } while (bn_cmp_dig(t0, 2) == CMP_LT);
347  /* t2 = a - 1. */
348  bn_copy(t2, a);
349  bn_sub_dig(t2, t2, 1);
350  /* t1 = (a - 1)/2. */
351  bn_rsh(t1, t2, 1);
352  /* t1 = t0^(a - 1)/2 mod a. */
353 #if BN_MOD != PMERS
354  bn_mxp(t1, t0, t1, a);
355 #else
356  bn_exp(t1, t0, t1, a);
357 #endif
358  /* If t1 != 1 and t1 != n - 1 return 0 */
359  if (bn_cmp_dig(t1, 1) != CMP_EQ && bn_cmp(t1, t2) != CMP_EQ) {
360  result = 0;
361  break;
362  }
363 
364  /* t2 = (t0|a). */
365  bn_smb_jac(t2, t0, a);
366  if (bn_sign(t2) == BN_NEG) {
367  bn_add(t2, t2, a);
368  }
369  /* If t1 != t2 (mod a) return 0. */
370  bn_mod(t1, t1, a);
371  bn_mod(t2, t2, a);
372  if (bn_cmp(t1, t2) != CMP_EQ) {
373  result = 0;
374  break;
375  }
376  }
377  }
378  CATCH_ANY {
379  result = 0;
380  THROW(ERR_CAUGHT);
381  }
382  FINALLY {
383  bn_free(t0);
384  bn_free(t1);
385  bn_free(t2);
386  }
387  return result;
388 }

407 void bn_gen_prime_safep(bn_t a, int bits) {
408  while (1) {
409  do {
410  bn_rand(a, BN_POS, bits);
411  } while (bn_bits(a) != bits);
412  /* Check if (a - 1)/2 is prime. */
413  bn_sub_dig(a, a, 1);
414  bn_rsh(a, a, 1);
415  if (bn_is_prime(a)) {
416  /* Restore a. */
417  bn_lsh(a, a, 1);
418  bn_add_dig(a, a, 1);
419  if (bn_is_prime(a)) {
420  /* Should be prime now. */
421  return;
422  }
423  }

431 void bn_gen_prime_stron(bn_t a, int bits) {
432  dig_t i, j;
433  int found, k;
434  bn_t r, s, t;
435 
436  bn_null(r);
437  bn_null(s);
438  bn_null(t);
439 
440  TRY {
441  bn_new(r);
442  bn_new(s);
443  bn_new(t);
444 
445  do {
446  do {
447  /* Generate two large primes r and s. */
448  bn_rand(s, BN_POS, bits / 2 - BN_DIGIT / 2);
449  bn_rand(t, BN_POS, bits / 2 - BN_DIGIT / 2);
450  } while (!bn_is_prime(s) || !bn_is_prime(t));
451  found = 1;
452  bn_rand(a, BN_POS, bits / 2 - bn_bits(t) - 1);
453  i = a->dp[0];
454  bn_dbl(t, t);
455  do {
456  /* Find first prime r = 2 * i * t + 1. */
457  bn_mul_dig(r, t, i);
458  bn_add_dig(r, r, 1);
459  i++;
460  if (bn_bits(r) > bits / 2 - 1) {
461  found = 0;
462  break;
463  }
464  } while (!bn_is_prime(r));
465  if (found == 0) {
466  continue;
467  }
468  /* Compute t = 2 * (s^(r-2) mod r) * s - 1. */
469  bn_sub_dig(t, r, 2);
470 #if BN_MOD == PMERS
471  bn_exp(t, s, t, r);
472 #else
473  bn_mxp(t, s, t, r);
474 #endif
475 
476  bn_mul(t, t, s);
477  bn_dbl(t, t);
478  bn_sub_dig(t, t, 1);
479 
480  k = bits - bn_bits(r);
481  k -= bn_bits(s);
482  bn_rand(a, BN_POS, k);
483  j = a->dp[0];
484  do {
485  /* Find first prime a = t + 2 * j * r * s. */
486  bn_mul(a, r, s);
487  bn_mul_dig(a, a, j);
488  bn_dbl(a, a);
489  bn_add(a, a, t);
490  j++;
491  if (bn_bits(a) > bits) {
492  found = 0;
493  break;
494  }
495  } while (!bn_is_prime(a));
496  } while (found == 0 && bn_bits(a) != bits);
497  }
498  CATCH_ANY {
499  THROW(ERR_CAUGHT);
500  }
501  FINALLY {
502  bn_free(r);
503  bn_free(s);
504  bn_free(t);
505  }
506 }
