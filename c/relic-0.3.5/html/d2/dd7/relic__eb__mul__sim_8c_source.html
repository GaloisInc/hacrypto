59 static void eb_mul_sim_kbltz(eb_t r, const eb_t p, const bn_t k, const eb_t q,
60  const bn_t m, const eb_t *t) {
61  int l0, l1, l, i, n0, n1, w, g;
62  signed char u, tnaf0[FB_BITS + 8], tnaf1[FB_BITS + 8], *_k, *_m;
63  eb_t t0[1 << (EB_WIDTH - 2)];
64  eb_t t1[1 << (EB_WIDTH - 2)];
65  bn_t vm, s0, s1;
66 
67  bn_null(vm);
68  bn_null(s0);
69  bn_null(s1);
70 
71  for (i = 0; i < (1 << (EB_WIDTH - 2)); i++) {
72  eb_null(t0[i]);
73  eb_null(t1[i]);
74  }
75 
76  TRY {
77  bn_new(vm);
78  bn_new(s0);
79  bn_new(s1);
80 
81  /* Compute the w-TNAF representation of k. */
82  if (eb_curve_opt_a() == OPT_ZERO) {
83  u = -1;
84  } else {
85  u = 1;
86  }
87 
88  g = (t == NULL ? 0 : 1);
89  if (!g) {
90  for (i = 0; i < (1 << (EB_WIDTH - 2)); i++) {
91  eb_new(t0[i]);
92  eb_set_infty(t0[i]);
93  fb_set_bit(t0[i]->z, 0, 1);
94  t0[i]->norm = 1;
95  }
96  eb_tab(t0, p, EB_WIDTH);
97  t = (const eb_t *)t0;
98  }
99 
100  /* Prepare the precomputation table. */
101  for (i = 0; i < (1 << (EB_WIDTH - 2)); i++) {
102  eb_new(t1[i]);
103  eb_set_infty(t1[i]);
104  fb_set_bit(t1[i]->z, 0, 1);
105  t1[i]->norm = 1;
106  }
107  /* Compute the precomputation table. */
108  eb_tab(t1, q, EB_WIDTH);
109 
110  /* Compute the w-TNAF representation of k. */
111  if (g) {
112  w = EB_DEPTH;
113  } else {
114  w = EB_WIDTH;
115  }
116  eb_curve_get_vm(vm);
117  eb_curve_get_s0(s0);
118  eb_curve_get_s1(s1);
119 
120  l0 = l1 = FB_BITS + 8;
121  bn_rec_tnaf(tnaf0, &l0, k, vm, s0, s1, u, FB_BITS, w);
122  bn_rec_tnaf(tnaf1, &l1, m, vm, s0, s1, u, FB_BITS, EB_WIDTH);
123 
124  l = MAX(l0, l1);
125  _k = tnaf0 + l - 1;
126  _m = tnaf1 + l - 1;
127  for (i = l0; i < l; i++)
128  tnaf0[i] = 0;
129  for (i = l1; i < l; i++)
130  tnaf1[i] = 0;
131 
132  _k = tnaf0 + l - 1;
133  _m = tnaf1 + l - 1;
134  eb_set_infty(r);
135  for (i = l - 1; i >= 0; i--, _k--, _m--) {
136  eb_frb(r, r);
137 
138  n0 = *_k;
139  n1 = *_m;
140  if (n0 > 0) {
141  eb_add(r, r, t[n0 / 2]);
142  }
143  if (n0 < 0) {
144  eb_sub(r, r, t[-n0 / 2]);
145  }
146  if (n1 > 0) {
147  eb_add(r, r, t1[n1 / 2]);
148  }
149  if (n1 < 0) {
150  eb_sub(r, r, t1[-n1 / 2]);
151  }
152  }
153  /* Convert r to affine coordinates. */
154  eb_norm(r, r);
155  }
156  CATCH_ANY {
157  THROW(ERR_CAUGHT);
158  }
159  FINALLY {
160  if (!g) {
161  for (i = 0; i < (1 << (EB_WIDTH - 2)); i++) {
162  eb_free(t0[i]);
163  }
164  }
165  for (i = 0; i < (1 << (EB_WIDTH - 2)); i++) {
166  eb_free(t1[i]);
167  }
168  bn_free(vm);
169  bn_free(s0);
170  bn_free(s1);
171  }
172 }

333 
334  eb_curve_get_ord(n);
335

337  eb_new(t0[i]);

341  eb_new(t[i]);

346  eb_add(t0[i], t0[i - 1], p);
347  }

368  for (int i = l0; i < l1; i++) {

419 void eb_mul_sim_joint(eb_t r, const eb_t p, const bn_t k, const eb_t q,
420  const bn_t m) {
421  eb_t t[5];
422  int u_i, len, offset;
423  signed char jsf[2 * (FB_BITS + 1)];
424  int i;
425 
426  eb_null(t[0]);
427  eb_null(t[1]);
428  eb_null(t[2]);
429  eb_null(t[3]);
430  eb_null(t[4]);
431 
432  TRY {
433  for (i = 0; i < 5; i++) {
434  eb_new(t[i]);
435  }
436 
437  eb_set_infty(t[0]);
438  eb_copy(t[1], q);
439  eb_copy(t[2], p);
440  eb_add(t[3], p, q);
441  eb_sub(t[4], p, q);
442 #if defined(EB_MIXED)
443  eb_norm_sim(t + 3, (const eb_t*)(t + 3), 2);
444 #endif
445 
446  len = 2 * (FB_BITS + 1);
447  bn_rec_jsf(jsf, &len, k, m);
448 
449  eb_set_infty(r);
450 
451  offset = MAX(bn_bits(k), bn_bits(m)) + 1;
452  for (i = len - 1; i >= 0; i--) {
453  eb_dbl(r, r);
454  if (jsf[i] != 0 && jsf[i] == -jsf[i + offset]) {
455  u_i = jsf[i] * 2 + jsf[i + offset];
456  if (u_i < 0) {
457  eb_sub(r, r, t[4]);
458  } else {
459  eb_add(r, r, t[4]);
460  }
461  } else {
462  u_i = jsf[i] * 2 + jsf[i + offset];
463  if (u_i < 0) {
464  eb_sub(r, r, t[-u_i]);
465  } else {
466  eb_add(r, r, t[u_i]);
467  }
468  }
469  }
470  eb_norm(r, r);
471  }
472  CATCH_ANY {
473  THROW(ERR_CAUGHT);
474  }
475  FINALLY {
476  for (i = 0; i < 5; i++) {
477  eb_free(t[i]);
478  }
479  }
480 }
