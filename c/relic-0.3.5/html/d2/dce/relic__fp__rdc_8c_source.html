77 void fp_rdc_monty_basic(fp_t c, dv_t a) {
78  int i;
79  dig_t r, c0, c1, *tmp, u0;
80  const dig_t *p = NULL;
81 
82  tmp = a;
83 
84  u0 = *(fp_prime_get_rdc());
85  p = fp_prime_get();
86 
87  c1 = 0;
88  for (i = 0; i < FP_DIGS; i++, tmp++) {
89  r = (dig_t)(*tmp * u0);
90  c0 = fp_mula_low(tmp, fp_prime_get(), r);
91  /* We must use this because the size (FP_DIGS - i) is variable. */
92  c1 += bn_add1_low(tmp + FP_DIGS, tmp + FP_DIGS, c0, FP_DIGS - i);
93  }
94  fp_copy(c, a + FP_DIGS);
95 
96  for (i = 0; i < c1; i++) {
97  fp_subn_low(c, c, p);
98  }
99  if (fp_cmpn_low(c, p) != CMP_LT) {
100  fp_subn_low(c, c, p);
101  }
102 }
