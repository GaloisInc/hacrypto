55 static void find_trace() {
56  fb_t t0, t1;
57  int counter;
58  ctx_t *ctx = core_get();
59 
60  fb_null(t0);
61  fb_null(t1);
62 
63  ctx->fb_ta = ctx->fb_tb = ctx->fb_tc = -1;
64 
65  TRY {
66  fb_new(t0);
67  fb_new(t1);
68 
69  counter = 0;
70  for (int i = 0; i < FB_BITS; i++) {
71  fb_zero(t0);
72  fb_set_bit(t0, i, 1);
73  fb_copy(t1, t0);
74  for (int j = 1; j < FB_BITS; j++) {
75  fb_sqr(t1, t1);
76  fb_add(t0, t0, t1);
77  }
78  if (!fb_is_zero(t0)) {
79  switch (counter) {
80  case 0:
81  ctx->fb_ta = i;
82  ctx->fb_tb = ctx->fb_tc = -1;
83  break;
84  case 1:
85  ctx->fb_tb = i;
86  ctx->fb_tc = -1;
87  break;
88  case 2:
89  ctx->fb_tc = i;
90  break;
91  default:
92  THROW(ERR_NO_VALID);
93  break;
94  }
95  counter++;
96  }
97  }
98  }
99  CATCH_ANY {
100  THROW(ERR_CAUGHT);
101  }
102  FINALLY {
103  fb_free(t0);
104  fb_free(t1);
105  }
106 }

117 static void find_solve() {
118  int i, j, k, l;
119  fb_t t0;
120  ctx_t *ctx = core_get();
121 
122  fb_null(t0);
123 
124  TRY {
125  fb_new(t0);
126 
127  l = 0;
128  for (i = 0; i < FB_BITS; i += 8, l++) {
129  for (j = 0; j < 16; j++) {
130  fb_zero(t0);
131  for (k = 0; k < 4; k++) {
132  if (j & (1 << k)) {
133  fb_set_bit(t0, i + 2 * k + 1, 1);
134  }
135  }
136  fb_copy(ctx->fb_half[l][j], t0);
137  for (k = 0; k < (FB_BITS - 1) / 2; k++) {
138  fb_sqr(ctx->fb_half[l][j], ctx->fb_half[l][j]);
139  fb_sqr(ctx->fb_half[l][j], ctx->fb_half[l][j]);
140  fb_add(ctx->fb_half[l][j], ctx->fb_half[l][j], t0);
141  }
142  }
143  fb_rsh(ctx->fb_half[l][j], ctx->fb_half[l][j], 1);
144  }
145  }
146  CATCH_ANY {
147  THROW(ERR_CAUGHT);
148  }
149  FINALLY {
150  fb_free(t0);
151  }
152 }

184 static void find_chain() {
185  int i, j, k, l;
186  ctx_t *ctx = core_get();
187 
188  ctx->chain_len = -1;
189  for (int i = 0; i < MAX_TERMS; i++) {
190  ctx->chain[i] = (i << 8) + i;
191  }
192  switch (FB_BITS) {
193  case 127:
194  ctx->chain[1] = (1 << 8) + 0;
195  ctx->chain[4] = (4 << 8) + 2;
196  ctx->chain[7] = (7 << 8) + 2;
197  ctx->chain_len = 9;
198  break;
199  case 193:
200  ctx->chain[1] = (1 << 8) + 0;
201  ctx->chain_len = 8;
202  break;
203  case 233:
204  ctx->chain[1] = (1 << 8) + 0;
205  ctx->chain[3] = (3 << 8) + 0;
206  ctx->chain[6] = (6 << 8) + 0;
207  ctx->chain_len = 10;
208  break;
209  case 251:
210  ctx->chain[1] = (1 << 8) + 0;
211  ctx->chain[2] = (2 << 8) + 1;
212  ctx->chain[4] = (4 << 8) + 3;
213  ctx->chain[5] = (5 << 8) + 4;
214  ctx->chain[7] = (7 << 8) + 6;
215  ctx->chain[8] = (8 << 8) + 7;
216  ctx->chain_len = 10;
217  break;
218  case 353:
219  ctx->chain[2] = (2 << 8) + 0;
220  ctx->chain[4] = (4 << 8) + 0;
221  ctx->chain_len = 10;
222  break;
223  case 367:
224  ctx->chain[1] = (1 << 8) + 0;
225  ctx->chain[2] = (2 << 8) + 1;
226  ctx->chain[6] = (6 << 8) + 3;
227  ctx->chain[9] = (9 << 8) + 2;
228  ctx->chain_len = 11;
229  break;
230  case 1223:
231  ctx->chain[1] = (1 << 8) + 0;
232  ctx->chain[2] = (2 << 8) + 0;
233  ctx->chain[4] = (4 << 8) + 2;
234  ctx->chain[5] = (5 << 8) + 4;
235  ctx->chain[10] = (10 << 8) + 2;
236  ctx->chain[11] = (11 << 8) + 10;
237  ctx->chain_len = 13;
238  break;
239  default:
240  l = 0;
241  j = (FB_BITS - 1);
242  for (k = 16; k >= 0; k--) {
243  if (j & (1 << k)) {
244  break;
245  }
246  }
247  for (i = 1; i < k; i++) {
248  if (j & (1 << i)) {
249  l++;
250  }
251  }
252  i = 0;
253  ctx->chain_len = k + l;
254  while (j != 1) {
255  if ((j & 0x01) != 0) {
256  i++;
257  ctx->chain[ctx->chain_len - i] = ((ctx->chain_len - i) << 8) + 0;
258  }
259  i++;
260  j = j >> 1;
261  }
262  break;
263  }
264 
265  int x, y, u[ctx->chain_len + 1];
266 
267  for (i = 0; i < MAX_TERMS; i++) {
268  for (j = 0; j < FB_TABLE; j++) {
269  ctx->fb_tab_ptr[i][j] = &(ctx->fb_tab_sqr[i][j]);
270  }
271  }
272 
273  u[0] = 1;
274  u[1] = 2;
275  for (i = 2; i <= ctx->chain_len; i++) {
276  x = ctx->chain[i - 1] >> 8;
277  y = ctx->chain[i - 1] - (x << 8);
278  if (x == y) {
279  u[i] = 2 * u[i - 1];
280  } else {
281  u[i] = u[x] + u[y];
282  }
283  }
284 
285  for (i = 0; i <= ctx->chain_len; i++) {
286  fb_itr_pre((fb_t *)fb_poly_tab_sqr(i), u[i]);
287  }
288 }

337 void fb_poly_add(fb_t c, const fb_t a) {
338  ctx_t *ctx = core_get();
339 
340  if (c != a) {
341  fb_copy(c, a);
342  }
343 
344  if (ctx->fb_pa != 0) {
345  c[FB_DIGS - 1] ^= ctx->fb_poly[FB_DIGS - 1];
346  if (ctx->fb_na != FB_DIGS - 1) {
347  c[ctx->fb_na] ^= ctx->fb_poly[ctx->fb_na];
348  }
349  if (ctx->fb_pb != 0 && ctx->fb_pc != 0) {
350  if (ctx->fb_nb != ctx->fb_na) {
351  c[ctx->fb_nb] ^= ctx->fb_poly[ctx->fb_nb];
352  }
353  if (ctx->fb_nc != ctx->fb_na && ctx->fb_nc != ctx->fb_nb) {
354  c[ctx->fb_nc] ^= ctx->fb_poly[ctx->fb_nc];
355  }
356  }
357  if (ctx->fb_na != 0 && ctx->fb_nb != 0 && ctx->fb_nc != 0) {
358  c[0] ^= 1;
359  }
360  } else {
361  fb_add(c, a, ctx->fb_poly);
362  }
363 }
