46 static void bn_add_imp(bn_t c, const bn_t a, const bn_t b) {
47  int max, min;
48  dig_t carry;
49 
50  max = a->used;
51  min = b->used;
52 
53  /* Grow the result. */
54  bn_grow(c, max);
55 
56  if (a->used == b->used) {
57  carry = bn_addn_low(c->dp, a->dp, b->dp, max);
58  } else {
59  carry = bn_addn_low(c->dp, a->dp, b->dp, min);
60  carry = bn_add1_low(c->dp + min, a->dp + min, carry, max - min);
61  }
62  if (carry) {
63  bn_grow(c, max + 1);
64  c->dp[max] = carry;
65  }
66  c->used = max + carry;
67  bn_trim(c);
68 }

101 void bn_add(bn_t c, const bn_t a, const bn_t b) {
102  int sa, sb;
103 
104  sa = a->sign;
105  sb = b->sign;
106 
107  if (sa == sb) {
108  /* If the signs are equal, copy the sign and add. */
109  c->sign = sa;
110  if (bn_cmp_abs(a, b) == CMP_LT) {
111  bn_add_imp(c, b, a);
112  } else {
113  bn_add_imp(c, a, b);
114  }
115  } else {
116  /* If the signs are different, subtract. */
117  if (bn_cmp_abs(a, b) == CMP_LT) {
118  c->sign = sb;
119  bn_sub_imp(c, b, a);
120  } else {
121  c->sign = sa;
122  bn_sub_imp(c, a, b);
123  }
124  }
125 }

127 void bn_add_dig(bn_t c, const bn_t a, dig_t b) {
128  dig_t carry;
129 
130  bn_grow(c, a->used);
131 
132  if (a->sign == BN_POS) {
133  carry = bn_add1_low(c->dp, a->dp, b, a->used);
134  if (carry) {
135  bn_grow(c, a->used + 1);
136  c->dp[a->used] = carry;
137  }
138  c->used = a->used + carry;
139  c->sign = BN_POS;
140  } else {
141  /* If a < 0 && |a| >= b, compute c = -(|a| - b). */
142  if (a->used > 1 || a->dp[0] >= b) {
143  carry = bn_sub1_low(c->dp, a->dp, b, a->used);
144  c->used = a->used;
145  c->sign = BN_NEG;
146  } else {
147  /* If a < 0 && |a| < b. */
148  if (a->used == 1) {
149  c->dp[0] = b - a->dp[0];
150  } else {
151  c->dp[0] = b;
152  }
153  c->used = 1;
154  c->sign = BN_POS;
155  }
156  }
157  bn_trim(c);
158 }

160 void bn_sub(bn_t c, const bn_t a, const bn_t b) {
161  int sa, sb;
162 
163  sa = a->sign;
164  sb = b->sign;
165 
166  if (sa != sb) {
167  /* If the signs are different, copy the sign of the first number and
168  * add. */
169  c->sign = sa;
170  if (bn_cmp_abs(a, b) == CMP_LT) {
171  bn_add_imp(c, b, a);
172  } else {
173  bn_add_imp(c, a, b);
174  }
175  } else {
176  /* If the signs are equal, adjust the sign and subtract. */
177  if (bn_cmp_abs(a, b) != CMP_LT) {
178  c->sign = sa;
179  bn_sub_imp(c, a, b);
180  } else {
181  c->sign = (sa == BN_POS) ? BN_NEG : BN_POS;
182  bn_sub_imp(c, b, a);
183  }
184  }
185 }
