52 static void fp_sqr_karat_imp(dv_t c, const fp_t a, int size, int level) {
53  int i, h, h1;
54  dv_t t0, t1, a0a0, a1a1;
55  dig_t carry;
56 
57  /* Compute half the digits of a or b. */
58  h = size >> 1;
59  h1 = size - h;
60 
61  dv_null(t0);
62  dv_null(t1);
63  dv_null(a0a0);
64  dv_null(a1a1);
65 
66  TRY {
67  /* Allocate the temp variables. */
68  dv_new(t0);
69  dv_new(t1);
70  dv_new(a0a0);
71  dv_new(a1a1);
72  dv_zero(t0, 2 * h1);
73  dv_zero(t1, 2 * (h1 + 1));
74  dv_zero(a0a0, 2 * h);
75  dv_zero(a1a1, 2 * h1);
76 
77  if (level <= 1) {
78  /* a0a0 = a0 * a0 and a1a1 = a1 * a1 */
79 #if FP_SQR == BASIC
80  for (i = 0; i < h; i++) {
81  bn_sqra_low(a0a0 + (2 * i), a + i, h - i);
82  }
83  for (i = 0; i < h1; i++) {
84  bn_sqra_low(a1a1 + (2 * i), a + h + i, h1 - i);
85  }
86 #elif FP_SQR == COMBA || FP_SQR == INTEG
87  bn_sqrn_low(a0a0, a, h);
88  bn_sqrn_low(a1a1, a + h, h1);
89 #elif FP_SQR == MULTP
90  bn_muln_low(a0a0, a, a, h);
91  bn_muln_low(a1a1, a + h, a + h, h1);
92 #endif
93  } else {
94  fp_sqr_karat_imp(a0a0, a, h, level - 1);
95  fp_sqr_karat_imp(a1a1, a + h, h1, level - 1);
96  }
97 
98  /* t2 = a1 * a1 << 2*h digits + a0 * a0. */
99  for (i = 0; i < 2 * h; i++) {
100  c[i] = a0a0[i];
101  }
102  for (i = 0; i < 2 * h1; i++) {
103  c[2 * h + i] = a1a1[i];
104  }
105 
106  /* t = (a1 + a0) */
107  carry = bn_addn_low(t0, a, a + h, h);
108  carry = bn_add1_low(t0 + h, t0 + h, carry, 2);
109  if (h1 > h) {
110  carry = bn_add1_low(t0 + h, t0 + h, *(a + 2 * h), 2);
111  }
112 
113  if (level <= 1) {
114  /* a1a1 = (a1 + a0)*(a1 + a0) */
115 #if FP_SQR == BASIC
116  for (i = 0; i < h1 + 1; i++) {
117  bn_sqra_low(t1 + (2 * i), t0 + i, h1 + 1 - i);
118  }
119 #elif FP_SQR == COMBA || FP_SQR == INTEG
120  bn_sqrn_low(t1, t0, h1 + 1);
121 #elif FP_SQR == MULTP
122  bn_muln_low(t1, t0, t0, h1 + 1);
123 #endif
124  } else {
125  fp_sqr_karat_imp(t1, t0, h1 + 1, level - 1);
126  }
127 
128  /* t = t - (a0*a0 << h digits) */
129  carry = bn_subn_low(t1, t1, a0a0, 2 * h);
130  bn_sub1_low(t1 + 2 * h, t1 + 2 * h, carry, 2 * (h1 + 1) - 2 * h);
131 
132  /* t = t - (a1*a1 << h digits) */
133  carry = bn_subn_low(t1, t1, a1a1, 2 * h1);
134  bn_sub1_low(t1 + 2 * h, t1 + 2 * h, carry, 2 * (h1 + 1) - 2 * h);
135 
136  /* c = c + [(a1 + a0)*(a1 + a0) << digits] */
137  c += h;
138  carry = bn_addn_low(c, c, t1, 2 * (h1 + 1));
139  c += 2 * (h1 + 1);
140  carry = bn_add1_low(c, c, carry, 2 * size - h - 2 * (h1 + 1));
141  }
142  CATCH_ANY {
143  THROW(ERR_CAUGHT);
144  }
145  FINALLY {
146  dv_free(t0);
147  dv_free(t1);
148  dv_free(a0a0);
149  dv_free(a1a1);
150  }
151 }
