39 void ep2_mul(ep2_t r, ep2_t p, bn_t k) {
40  int i, l;
41  ep2_t t;
42 
43  ep2_null(t);
44  TRY {
45  ep2_new(t);
46  l = bn_bits(k);
47 
48  if (bn_test_bit(k, l - 1)) {
49  ep2_copy(t, p);
50  } else {
51  ep2_set_infty(t);
52  }
53 
54  for (i = l - 2; i >= 0; i--) {
55  ep2_dbl(t, t);
56  if (bn_test_bit(k, i)) {
57  ep2_add(t, t, p);
58  }
59  }
60 
61  ep2_copy(r, t);
62  ep2_norm(r, r);
63  }
64  CATCH_ANY {
65  THROW(ERR_CAUGHT);
66  }
67  FINALLY {
68  ep2_free(t);
69  }
70 }
