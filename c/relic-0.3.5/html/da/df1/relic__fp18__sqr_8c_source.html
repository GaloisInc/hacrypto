42 void fp18_sqr_basic(fp18_t c, fp18_t a) {
43  fp6_t t0, t1, t2, t3, t4;
44 
45  fp6_null(t0);
46  fp6_null(t1);
47  fp6_null(t2);
48  fp6_null(t3);
49  fp6_null(t4);
50 
51  TRY {
52  fp6_new(t0);
53  fp6_new(t1);
54  fp6_new(t2);
55  fp6_new(t3);
56  fp6_new(t4);
57 
58  /* t0 = a_0^2 */
59  fp6_sqr(t0, a[0]);
60 
61  /* t1 = 2 * a_1 * a_2 */
62  fp6_mul(t1, a[1], a[2]);
63  fp6_dbl(t1, t1);
64 
65  /* t2 = a_2^2. */
66  fp6_sqr(t2, a[2]);
67 
68  /* c_2 = a_0 + a_2. */
69  fp6_add(c[2], a[0], a[2]);
70 
71  /* t3 = (a_0 + a_2 + a_1)^2. */
72  fp6_add(t3, c[2], a[1]);
73  fp6_sqr(t3, t3);
74 
75  /* c_2 = (a_0 + a_2 - a_1)^2. */
76  fp6_sub(c[2], c[2], a[1]);
77  fp6_sqr(c[2], c[2]);
78 
79  /* c_2 = (c_2 + t3)/2. */
80  fp6_add(c[2], c[2], t3);
81  for (int i = 0; i < 2; i++) {
82  fp_hlv(c[2][0][i], c[2][0][i]);
83  fp_hlv(c[2][1][i], c[2][1][i]);
84  fp_hlv(c[2][2][i], c[2][2][i]);
85  }
86 
87  /* t3 = t3 - c_2 - t1. */
88  fp6_sub(t3, t3, c[2]);
89  fp6_sub(t3, t3, t1);
90 
91  /* c_2 = c_2 - t0 - t2. */
92  fp6_sub(c[2], c[2], t0);
93  fp6_sub(c[2], c[2], t2);
94 
95  /* c_0 = t0 + t1 * E. */
96  fp6_mul_art(t4, t1);
97  fp6_add(c[0], t0, t4);
98 
99  /* c_1 = t3 + t2 * E. */
100  fp6_mul_art(t4, t2);
101  fp6_add(c[1], t3, t4);
102  } CATCH_ANY {
103  THROW(ERR_CAUGHT);
104  } FINALLY {
105  fp6_free(t0);
106  fp6_free(t1);
107  fp6_free(t2);
108  fp6_free(t3);
109  fp6_free(t4);
110  }
111 }

138 
139  /* u1 = 2 * a_1 * a_2 */
140  fp6_mul_unr(u1, a[1], a[2]);
141  for (int i = 0; i < 3; i++) {
142  fp2_addc_low(u1[i], u1[i], u1[i]);
143  }
144 
145  /* t2 = a_2^2. */
146  fp6_sqr_unr(u2, a[2]);
147

149  fp6_add(c[2], a[0], a[2]);
150

152  fp6_add(t0, c[2], a[1]);
153  fp6_sqr_unr(u3, t0);
154 
155  /* u4 = (a_0 + a_2 - a_1)^2. */
156  fp6_sub(c[2], c[2], a[1]);
157  fp6_sqr_unr(u4, c[2]);
158 
159  /* u4 = (c_2 + t3)/2. */
160  for (int i = 0; i < 3; i++) {
161  fp2_addc_low(u4[i], u4[i], u3[i]);
162  fp_hlvd_low(u4[i][0], u4[i][0]);
163  fp_hlvd_low(u4[i][1], u4[i][1]);
164  }
165 
166  /* t3 = t3 - u4 - t1. */
167  for (int i = 0; i < 3; i++) {
168  fp2_subc_low(u3[i], u3[i], u4[i]);
169  fp2_subc_low(u3[i], u3[i], u1[i]);
170  }
171 
172  /* c_2 = u4 - t0 - t2. */
173  for (int i = 0; i < 3; i++) {
174  fp2_subc_low(u4[i], u4[i], u0[i]);
175  fp2_subc_low(u4[i], u4[i], u2[i]);
176  fp2_rdcn_low(c[2][i], u4[i]);
177  }
178 
179  /* c_0 = t0 + t1 * E. */
180  fp2_nord_low(u4[0], u1[2]);
181  fp2_addc_low(u1[2], u4[0], u0[0]);
182  fp2_rdcn_low(c[0][0], u1[2]);
183  fp2_addc_low(u1[2], u1[0], u0[1]);
184  fp2_rdcn_low(c[0][1], u1[2]);
185  fp2_addc_low(u1[2], u1[1], u0[2]);
186  fp2_rdcn_low(c[0][2], u1[2]);
187 
188  /* c_1 = t3 + t2 * E. */
189  fp2_nord_low(u4[0], u2[2]);
190  fp2_addc_low(u1[0], u4[0], u3[0]);
191  fp2_rdcn_low(c[1][0], u1[0]);
192  fp2_addc_low(u1[1], u2[0], u3[1]);
193  fp2_rdcn_low(c[1][1], u1[1]);
194  fp2_addc_low(u1[2], u2[1], u3[2]);
195  fp2_rdcn_low(c[1][2], u1[2]);
196  } CATCH_ANY {
197  THROW(ERR_CAUGHT);
198  } FINALLY {
199  dv6_free(u0);
200  dv6_free(u1);
201  dv6_free(u2);
202  dv6_free(u3);
203  dv6_free(u4);
204  fp6_free(t0);
205  }
206 }
