41 void fp2_sqr_basic(fp2_t c, fp2_t a) {
42  fp_t t0, t1, t2;
43 
44  fp_null(t0);
45  fp_null(t1);
46  fp_null(t2);
47 
48  TRY {
49  fp_new(t0);
50  fp_new(t1);
51  fp_new(t2);
52 
53  /* t0 = (a_0 + a_1). */
54  fp_add(t0, a[0], a[1]);
55 
56  /* t1 = (a_0 - a_1). */
57  fp_sub(t1, a[0], a[1]);
58 
59  /* t1 = a_0 + u^2 * a_1. */
60  for (int i = -1; i > fp_prime_get_qnr(); i--) {
61  fp_sub(t1, t1, a[1]);
62  }
63 
64  if (fp_prime_get_qnr() == -1) {
65  /* t2 = 2 * a_0. */
66  fp_dbl(t2, a[0]);
67  /* c_1 = 2 * a_0 * a_1. */
68  fp_mul(c[1], t2, a[1]);
69  /* c_0 = a_0^2 + a_1^2 * u^2. */
70  fp_mul(c[0], t0, t1);
71  } else {
72  /* c_1 = a_0 * a_1. */
73  fp_mul(c[1], a[0], a[1]);
74  /* c_0 = a_0^2 + a_1^2 * u^2. */
75  fp_mul(c[0], t0, t1);
76  for (int i = -1; i > fp_prime_get_qnr(); i--) {
77  fp_add(c[0], c[0], c[1]);
78  }
79  /* c_1 = 2 * a_0 * a_1. */
80  fp_dbl(c[1], c[1]);
81  }
82  /* c = c_0 + c_1 * u. */
83  }
84  CATCH_ANY {
85  THROW(ERR_CAUGHT);
86  }
87  FINALLY {
88  fp_free(t0);
89  fp_free(t1);
90  fp_free(t2);
91  }
92 }
