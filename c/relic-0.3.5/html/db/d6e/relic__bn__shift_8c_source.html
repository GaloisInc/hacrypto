39 void bn_dbl(bn_t c, const bn_t a) {
40  dig_t carry;
41 
42  bn_grow(c, a->used + 1);
43 
44  c->used = a->used;
45  carry = bn_lsh1_low(c->dp, a->dp, c->used);

66 void bn_lsh(bn_t c, const bn_t a, int bits) {
67  int digits;
68  dig_t carry;
69 
70  bn_copy(c, a);
71  if (bits <= 0) {
72  return;
73  }
74 
75  SPLIT(bits, digits, bits, BN_DIG_LOG);
76 
77  if (bits > 0) {
78  if (bn_bits(c) + bits > c->used * (int)BN_DIGIT) {
79  bn_grow(c, c->used + digits + 1);
80  }
81  } else {
82  bn_grow(c, c->used + digits);
83  }
84 
85  if (digits > 0) {
86  bn_lshd_low(c->dp, a->dp, a->used, digits);
87  }
88  c->used = a->used + digits;
89  c->sign = a->sign;
90 
91  if (bits > 0) {
92  if (c != a) {
93  carry = bn_lshb_low(c->dp + digits, a->dp, a->used, bits);
94  } else {
95  carry = bn_lshb_low(c->dp + digits, c->dp + digits, c->used - digits, bits);
96  }
97  if (carry != 0) {
98  c->dp[c->used] = carry;
99  (c->used)++;
100  }
101  }
102  bn_trim(c);
103 }

105 void bn_rsh(bn_t c, const bn_t a, int bits) {
106  int digits = 0;
107 
108  if (bits <= 0) {
109  bn_copy(c, a);
110  return;
111  }
112 
113  SPLIT(bits, digits, bits, BN_DIG_LOG);
114 
115  if (digits > 0) {
116  bn_rshd_low(c->dp, a->dp, a->used, digits);
117  }
118  c->used = a->used - digits;
119  c->sign = a->sign;
120 
121  if (c->used > 0 && bits > 0) {
122  if (digits == 0 && c != a) {
123  bn_rshb_low(c->dp, a->dp + digits, a->used - digits, bits);
124  } else {
