47 static void pp_exp_bn(fp12_t c, fp12_t a) {
48  fp12_t t0, t1, t2, t3;
49  int l = MAX_TERMS + 1, b[MAX_TERMS + 1];
50  bn_t x;
51 
52  fp12_null(t0);
53  fp12_null(t1);
54  fp12_null(t2);
55  fp12_null(t3);
56  bn_null(x);
57 
58  TRY {
59  fp12_new(t0);
60  fp12_new(t1);
61  fp12_new(t2);
62  fp12_new(t3);
63  bn_new(x);
64 
65  /*
66  * New final exponentiation following Fuentes-Castañeda, Knapp and
67  * Rodríguez-Henríquez: Fast Hashing to G_2.
68  */
69  fp_param_get_var(x);
70  fp_param_get_sps(b, &l);
71 
72  /* First, compute m = f^(p^6 - 1)(p^2 + 1). */
73  fp12_conv_cyc(c, a);
74 
75  /* Now compute m^((p^4 - p^2 + 1) / r). */
76  /* t0 = m^2x. */
77  fp12_exp_cyc_sps(t0, c, b, l);
78  fp12_sqr_cyc(t0, t0);
79  /* t1 = m^6x. */
80  fp12_sqr_cyc(t1, t0);
81  fp12_mul(t1, t1, t0);
82  /* t2 = m^6x^2. */
83  fp12_exp_cyc_sps(t2, t1, b, l);
84  /* t3 = m^12x^3. */
85  fp12_sqr_cyc(t3, t2);
86  fp12_exp_cyc_sps(t3, t3, b, l);
87 
88  if (bn_sign(x) == BN_NEG) {
89  fp12_inv_uni(t0, t0);
90  fp12_inv_uni(t1, t1);
91  fp12_inv_uni(t3, t3);
92  }
93 
94  /* t3 = a = m^12x^3 * m^6x^2 * m^6x. */
95  fp12_mul(t3, t3, t2);
96  fp12_mul(t3, t3, t1);
97 
98  /* t0 = b = 1/(m^2x) * t3. */
99  fp12_inv_uni(t0, t0);
100  fp12_mul(t0, t0, t3);
101 
102  /* Compute t2 * t3 * m * b^p * a^p^2 * [b * 1/m]^p^3. */
103  fp12_mul(t2, t2, t3);
104  fp12_mul(t2, t2, c);
105  fp12_inv_uni(c, c);
106  fp12_mul(c, c, t0);
107  fp12_frb(c, c, 3);
108  fp12_mul(c, c, t2);
109  fp12_frb(t0, t0, 1);
110  fp12_mul(c, c, t0);
111  fp12_frb(t3, t3, 2);
112  fp12_mul(c, c, t3);
113  }
114  CATCH_ANY {
115  THROW(ERR_CAUGHT);
116  }
117  FINALLY {
118  fp12_free(t0);
119  fp12_free(t1);
120  fp12_free(t2);
121  fp12_free(t3);
122  bn_free(x);
123  }
124 }

133 static void pp_exp_b12(fp12_t c, fp12_t a) {
134  fp12_t t[10];
135  bn_t x;
136  int l = MAX_TERMS + 1, b[MAX_TERMS + 1];
137 
138  bn_null(x);
139 
140  TRY {
141  for (int i = 0; i < 10; i++) {
142  fp12_null(t[i]);
143  fp12_new(t[i]);
144  }
145  bn_new(x);
146 
147  fp_param_get_var(x);
148  fp_param_get_sps(b, &l);
149 
150  /* First, compute m^(p^6 - 1)(p^2 + 1). */
151  fp12_conv_cyc(c, a);
152 
153  /* v0 = f^-1. */
154  fp12_inv_uni(t[0], c);
155 
156  /* v1 = f^-2. */
157  fp12_sqr_cyc(t[1], t[0]);
158 
159  /* v2 = f^x. */
160  fp12_exp_cyc_sps(t[2], c, b, l);
161 
162  if (bn_sign(x) == BN_NEG) {
163  fp12_inv_uni(t[2], t[2]);
164  }
165 
166  /* v3 = f^2x. */
167  fp12_sqr_cyc(t[3], t[2]);
168 
169  /* v4 = f^(x - 2). */
170  fp12_mul(t[4], t[2], t[1]);
171 
172  /* v5 = f^(x^2 - 2x). */
173  fp12_exp_cyc_sps(t[5], t[4], b, l);
174 
175  if (bn_sign(x) == BN_NEG) {
176  fp12_inv_uni(t[5], t[5]);
177  }
178 
179  /* v6 = f^(x^3 - 2x^2). */
180  fp12_exp_cyc_sps(t[6], t[5], b, l);
181 
182  if (bn_sign(x) == BN_NEG) {
183  fp12_inv_uni(t[6], t[6]);
184  }
185 
186  /* v7 = f^(x^4 - 2x^3 + 2x). */
187  fp12_exp_cyc_sps(t[7], t[6], b, l);
188  if (bn_sign(x) == BN_NEG) {
189  fp12_inv_uni(t[7], t[7]);
190  }
191  fp12_mul(t[7], t[7], t[3]);
192 
193  /* v8 = f^(x^5 - 2x^4 + 2x^2). */
194  fp12_exp_cyc_sps(t[8], t[7], b, l);
195  if (bn_sign(x) == BN_NEG) {
196  fp12_inv_uni(t[8], t[8]);
197  }
198 
199  /* v7 = f^(x^4 - 2x^3 + 2x - 1)^p. */
200  fp12_mul(t[7], t[7], t[0]);
201  fp12_frb(t[7], t[7], 1);
202 
203  /* v6 = f^(x^3 - 2x^2 + x)^p^2. */
204  fp12_mul(t[6], t[6], t[2]);
205  fp12_frb(t[6], t[6], 2);
206 
207  /* v5 = f^(x^2 - 2x + 1)^p^3. */
208  fp12_mul(t[5], t[5], c);
209  fp12_frb(t[5], t[5], 1);
210  fp12_frb(t[5], t[5], 2);
211 
212  /* v4 = f^(2 - x). */
213  fp12_inv_uni(t[4], t[4]);
214 
215  /* Now compute f * v4 * v5 * v6 * v7 * v8. */
216  fp12_mul(c, c, t[4]);
217  fp12_mul(c, c, t[5]);
218  fp12_mul(c, c, t[6]);
219  fp12_mul(c, c, t[7]);
220  fp12_mul(c, c, t[8]);
221  }
222  CATCH_ANY {
223  THROW(ERR_CAUGHT);
224  }
225  FINALLY {
226  for (int i = 0; i < 9; i++) {
227  fp12_free(t[i]);
228  }
229  bn_free(x);
230  }
231 }

237 void pp_exp_k2(fp2_t c, fp2_t a) {
238  bn_t e, n;
239 
240  bn_null(n);
241  bn_null(e);
242 
243  TRY {
244  bn_new(n);
245  bn_new(e);
246 
247  ep_curve_get_ord(n);
248 
249  fp2_conv_uni(c, a);
250  dv_copy(e->dp, fp_prime_get(), FP_DIGS);
251  e->used = FP_DIGS;
252  e->sign = BN_POS;
253  bn_add_dig(e, e, 1);
254  bn_div(e, e, n);
255  fp2_exp_uni(c, c, e);
256  } CATCH_ANY {
257  THROW(ERR_CAUGHT);
258  } FINALLY {
259  bn_free(n);
260  bn_free(e);
261  }
262 }
